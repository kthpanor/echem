import matplotlib.cm as cm
import matplotlib.pyplot as plt
import numpy as np

a = 0.3
d = 1
T = 5.0

x = np.arange(-10, 10, 0.1)
y = np.arange(-10, 10, 0.1)
X, Y = np.meshgrid(x, y)

fig = plt.figure(1, figsize=(8, 4))

h1 = plt.subplot(2, 4, 1)
Z = np.exp(-a * ((X + d) ** 2 + Y**2)) + np.exp(-((X - d) ** 2 + Y**2))
plt.contour(X, Y, Z, linewidths=0.25)
plt.arrow(0, 0, 2.5, 0, head_width=0.5, head_length=0.5, fc="k", ec="k")
plt.arrow(0, 0, 0, 2.5, head_width=0.5, head_length=0.5, fc="k", ec="k")
plt.setp(
    h1,
    xlim=(-9, 9),
    ylim=(-8, 8),
    xticks=([-5, 0, 5]),
    xticklabels=[],
    yticks=([-5, 0, 5]),
    yticklabels=[r"$-a$", r"$0$", r"$a$"],
)
